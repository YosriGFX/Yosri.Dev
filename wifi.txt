#!/bin/bash


apt-get update -y
apt-get install -y dhcpcd hostapd dnsmasq git iptables

echo "options rtl8188eu rtw_power_mgnt=0 rtw_enusbss=0" | tee /etc/modprobe.d/rtl8188eu.conf
printf "interface wlan1
    static ip_address=10.42.0.1/24
    static routers=10.42.0.1" > /etc/dhcpcd.conf

printf "interface=wlan1
hw_mode=g
channel=6
ieee80211n=1
wmm_enabled=1
macaddr_acl=0
ignore_broadcast_ssid=0
auth_algs=1
wpa=2
wpa_key_mgmt=WPA-PSK
rsn_pairwise=CCMP
ssid=Home
wpa_passphrase=Home123456" > /etc/hostapd/hostapd.conf

printf '\nDAEMON_CONF="/etc/hostapd/hostapd.conf"\n\n' > /etc/default/hostapd

printf "interface=wlan1
listen-address=10.42.0.1
server=8.8.8.8
bind-interfaces
dhcp-range=10.42.0.2,10.42.0.20,12h" > /etc/dnsmasq.conf
echo "Creating /etc/sysctl.conf"
printf "net.ipv4.ip_forward=1" >> /etc/sysctl.conf
sh -c "echo 1 > /proc/sys/net/ipv4/ip_forward"

iptables -t nat -A POSTROUTING -o wlan0 -j MASQUERADE
iptables -A FORWARD -m conntrack --ctstate RELATED,ESTABLISHED -j ACCEPT
iptables -A FORWARD -i wlan1 -o wlan0 -j ACCEPT

sh -c "iptables-save > /etc/iptables.ipv4.nat"
printf "iptables-restore < /etc/iptables.ipv4.nat" >> /etc/rc.local
systemctl unmask hostapd
systemctl enable hostapd
service hostapd start
service dnsmasq start
